def initialize(tagName, markup, tokens)
  super

  if markup =~ Syntax then
    @id = $1

    if $2.nil? then
        @width = 560
        @height = 420
    else
        @width = $2.to_i
        @height = $3.to_i
    end
  else
    raise "No Vimeo ID provided in the \"youtube\" tag"
  end
end
